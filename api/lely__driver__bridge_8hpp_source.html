91  public:
92  COData data;
93  LelyDriverBridge *driver;
94  std::mutex mtx;
95  explicit TPDOWriteTask(ev_exec_t *exec)
96  : ev::CoTask(exec)
97  {
98  // Lock and load
99  mtx.lock();
100  }
101  void operator()() noexcept
102  {
103  std::scoped_lock<util::BasicLockable> lk(driver->tpdo_event_mutex);
104  switch (data.type_)
105  {
106  case CODataTypes::COData8:
107  driver->tpdo_mapped[data.index_][data.subindex_] =
108  static_cast<uint8_t>(data.data_);
109  break;
110  case CODataTypes::COData16:
111  driver->tpdo_mapped[data.index_][data.subindex_] =
112  static_cast<uint16_t>(data.data_);
113  break;
114  case CODataTypes::COData32:
115  driver->tpdo_mapped[data.index_][data.subindex_] =
116  static_cast<uint32_t>(data.data_);
117  break;
118  default:
119  break;
120  }
121  driver->master.TpdoWriteEvent(driver->id(), data.index_, data.subindex_);
122  // Unlock when done
123  mtx.unlock();
124  }
125  };
